# Comandos Linux

```bash
ls -l  # Lista arquivos e diretórios com detalhes
cd Doc<TAB>  # Auto completa e acessa o diretório 'Doc'
pwd  # Mostra o diretório atual
mkdir Aluno  # Cria um diretório chamado 'Aluno'
cd Aluno  # Acessa o diretório 'Aluno'
man pwd  # Abre o manual do comando 'pwd'
gedit  # Editor de texto gráfico simples
nano  # Editor de texto no terminal
vim  # Editor de texto avançado no terminal
gedit novo.txt  # Cria ou edita o arquivo 'novo.txt'
cp novo.txt copia.txt  # Copia 'novo.txt' para 'copia.txt'
mv copia.txt texto.txt  # Renomeia 'copia.txt' para 'texto.txt'
ls -l  # Lista arquivos e diretórios com detalhes
rm novo.txt  # Remove o arquivo 'novo.txt'
rm *.txt  # Remove todos os arquivos com extensão .txt
cd ..  # Retorna ao diretório anterior
pwd  # Confirma o diretório atual
strace pwd
ps -aux
gcc -o arq aqr.c # cria o executavel
ls
./arq
